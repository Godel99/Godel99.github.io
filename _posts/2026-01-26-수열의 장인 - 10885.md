[문제 링크](https://www.acmicpc.net/problem/10885) 

## **풀이**

### 문제 요약
문제를 간단히 요약하면 -2, -1, 0, 1, 2로만 구성된 수열의 구간 곱의 최댓값을 구하는 것입니다.

### 접근
수열의 구간 곱의 최댓값을 구하는 것이므로 일단 0은 제외하고 생각해야 합니다. 0을 기준으로 구간을 나눌 수 있겠습니다. 문제는 음수의 처리입니다.

음수의 개수가 홀수이냐 짝수이냐를 구해서 그리디하게 최댓값을 찾아갈 수도 있겠지만, 연산이 좀 더 복잡해집니다.

### 구간 늘리기
어차피 0 제외하고는 구간이 늘어날수록 최댓값이 커지는 구조입니다. 

그렇다면 최대한 긴 구간을 잡기 위해 **왼쪽에서 첫 음수를 제외하고 모든 오른쪽 구간(n->0)을 계산하고, 반대로 마지막 음수를 제외하고 모든 왼쪽 구간(0->n)을 계산**해 둘 중 큰 것을 최댓값으로 결정하면 됩니다. 제외한 음수는 결국 양쪽 끝에서 계산이 이뤄지므로 최댓값을 결정하는데 문제가 없습니다.

```python
import sys
def print(*args, sep=" ", end="\n"): sys.stdout.write(sep.join(map(str, args)) + end)
def input(): return sys.stdin.readline().rstrip()

def main():
    MOD = 1_000_000_007
    T = int(input())
    for _ in range(T):
        n = int(input())
        A = list(map(int, input().split()))
        mx = -3; mxcnt = -1; cnt = [0,0]; minus = minustwo = 0; 
        flag = 0
        for a in A:
            mx = max(mx, a)
            if a > 0: flag = 1
            if a > 1: cnt[minus] += 1
            if a < 0: minus += 1
            if a < -1: minustwo += 1
            if minus == 2:
                flag = 1
                cnt[0] += minustwo+cnt[1]
                cnt[1] = minustwo = minus = 0
            if flag: mxcnt = max(mxcnt, cnt[0], cnt[1])
            if not a: flag = cnt[0] = cnt[1] = minus = minustwo = 0
        if mxcnt < 0: print(mx); continue
        cnt[0] = cnt[1] = minus = minustwo = 0
        for a in reversed(A):
            if a > 0: flag = 1
            if a > 1: cnt[minus] += 1
            if a < 0: minus += 1
            if a < -1: minustwo += 1
            if minus == 2:
                flag = 1
                cnt[0] += minustwo+cnt[1]
                cnt[1] = minustwo = minus = 0
            if flag: mxcnt = max(mxcnt, cnt[0], cnt[1])
            if not a: flag = cnt[0] = cnt[1] = minus = minustwo = 0
        print(pow(2, mxcnt, MOD))
if __name__ == '__main__':
    sys.exit(main())
```

```c++
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
const ll MOD = 1'000'000'007;

ll pw(ll n, ll r){
    ll ret = 1;
    while(r){
        if(r&1) ret = ret*n%MOD;
        n = n*n%MOD;
        r>>=1;
    }
    return ret;
}

int main(){
    cin.tie(0); cout.tie(0); ios::sync_with_stdio(false);
    int T; cin >> T;
    while(T--){
        int n; cin >> n;
        vector<int> A(n);
        int mx = -3, mxcnt = -1, cnt[2] = {0, 0}, minus = 0, minustwo = 0;
        bool flag = 0;
        for(int i = 0; i < n; i++){
            cin >> A[i]; int x = A[i];
            mx = max(mx, x);
            if(x > 0) flag = 1;
            if(x > 1) cnt[minus]++;
            if(x < 0) minus++;
            if(x < -1) minustwo++;
            if(minus == 2){
                flag = 1;
                cnt[0] += minustwo+cnt[1];
                cnt[1] = minustwo = minus = 0;
            }
            if(flag) mxcnt = max({mxcnt, cnt[0], cnt[1]});
            if(!x) flag = cnt[0] = cnt[1] = minus = minustwo = 0;
        }
        if(mxcnt < 0){
            cout << mx << '\n'; continue;
        }
        cnt[0] = cnt[1] = minus = minustwo = 0;
        for(int i = 0; i < n; i++){
            int x = A[n-i-1];
            if(x > 0) flag = 1;
            if(x > 1) cnt[minus]++;
            if(x < 0) minus++;
            if(x < -1) minustwo++;
            if(minus == 2){
                flag = 1;
                cnt[0] += minustwo+cnt[1];
                cnt[1] = minustwo = minus = 0;
            }
            if(flag) mxcnt = max({mxcnt, cnt[0], cnt[1]});
            if(!x) flag = cnt[0] = cnt[1] = minus = minustwo = 0;
        }
        cout << pw(2, mxcnt) << '\n';
    }
    return 0;
}
```
